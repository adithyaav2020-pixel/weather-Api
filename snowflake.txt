AUTO_SUSPEND = 300

create schema weatherschema;

CREATE OR REPLACE TABLE weather_table (
    city STRING,
    temperature FLOAT,
    humidity FLOAT,
    condition STRING,
    timestamp TIMESTAMP_NTZ
);

CREATE OR REPLACE PIPE weather_pipe
  AUTO_INGEST = TRUE
AS
COPY INTO weather_table
FROM (
  SELECT 
    $1:city::string,
    $1:temperature::float,
    $1:humidity::float,    
    $1:condition::string,   
    $1:timestamp::timestamp_ntz
  FROM @weather_stage
)
FILE_FORMAT = (TYPE = 'JSON');

list @weather_stage;

select system$pipe_status('weather_pipe');

truncate table weather_table;
